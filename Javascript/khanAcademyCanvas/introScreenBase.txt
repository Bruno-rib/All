var t = 0;
var n = 1;


draw = function() {
    
    noStroke();
    field();
    pushMatrix();
    
    rotate(-45);
    translate(-100, 0);
    textSize(32);
    fill(random(100, 255), 0, 0, t/2 - 200);
    text ("BASEBALL MADNESS", 0, 300, 400, 200);
    popMatrix();
    buttons();

    if (t>300 && n>0.4) {
        n -= 0.01;
    }
    pushMatrix();
    scale(n);
    baseball();
    popMatrix();
    

    

    
    if (t<1000) {
        t++;
    }
    
};
